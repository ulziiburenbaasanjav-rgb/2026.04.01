#property strict
#include <Trade\Trade.mqh>

input group "=== BUY RSI Түвшин ба Эрсдэл (10, 15, 20, 25) ==="
input bool    InpAllowBuy      = true;
input double  InpBuy_L1        = 10.0; // Reset & 1st Entry
input int     InpBuy_L1_SL     = 100;  // 1-р оролтын SL (Pips)
input int     InpBuy_L1_TP     = 150;  // 1-р оролтын TP (Pips)

input double  InpBuy_L2        = 15.0;
input int     InpBuy_L2_SL     = 100;
input int     InpBuy_L2_TP     = 150;

input double  InpBuy_L3        = 20.0;
input int     InpBuy_L3_SL     = 100;
input int     InpBuy_L3_TP     = 150;

input double  InpBuy_L4        = 25.0;
input int     InpBuy_L4_SL     = 100;
input int     InpBuy_L4_TP     = 150;

input double  InpBuy_BE_Zone   = 30.0; // Buy-ийн SL-ийг Entry рүү зөөх RSI

input group "=== SELL RSI Түвшин ба Эрсдэл (90, 85, 80, 75) ==="
input bool    InpAllowSell     = true;
input double  InpSell_L1       = 90.0; // Reset & 1st Entry
input int     InpSell_L1_SL    = 100;  // 1-р оролтын SL (Pips)
input int     InpSell_L1_TP    = 150;  // 1-р оролтын TP (Pips)

input double  InpSell_L2       = 85.0;
input int     InpSell_L2_SL    = 100;
input int     InpSell_L2_TP    = 150;

input double  InpSell_L3       = 80.0;
input int     InpSell_L3_SL    = 100;
input int     InpSell_L3_TP    = 150;

input double  InpSell_L4       = 75.0;
input int     InpSell_L4_SL    = 100;
input int     InpSell_L4_TP    = 150;

input double  InpSell_BE_Zone  = 70.0; // Sell-ийн SL-ийг Entry рүү зөөх RSI

input group "=== Break Even (BE) Ашгийн Тохиргоо (Pips) ==="
input int     InpBE_Main_Pips  = 100;  // L1 (10/90) оролтуудын BE ашиг
input int     InpBE_Sub_Pips   = 50;   // Бусад оролтуудын BE ашиг
input int     InpBE_Add_Pips   = 2;    // Entry-гээс хэдэн pip дээр SL тавих

input group "=== Дансны удирдлага (Compound) ==="
input double InpBaseBalance    = 50.0;
input double InpBaseLot        = 0.03;
input double InpMaxLot         = 10.0;

// Magic Numbers
#define M_B1 987010
#define M_B2 987015
#define M_B3 987020
#define M_B4 987025
#define M_S1 988090
#define M_S2 988085
#define M_S3 988080
#define M_S4 988075

CTrade  trade;
int     handleRSI;
double  pipsToPoints;
bool    isBuyReady = false, isSellReady = false;
bool    b1, b2, b3, b4;
bool    s1, s2, s3, s4;

int OnInit() {
   pipsToPoints = (_Digits == 3 || _Digits == 5) ? 10.0 : 1.0;
   handleRSI = iRSI(_Symbol, PERIOD_M5, 14, PRICE_CLOSE);
   return(INIT_SUCCEEDED);
}

void OnTick() {
   double rsiBuffer[];
   ArraySetAsSeries(rsiBuffer, true);
   if(CopyBuffer(handleRSI, 0, 0, 2, rsiBuffer) < 2) return;
   double currRSI = rsiBuffer[0];

   // --- BUY LOGIC ---
   if(currRSI < InpBuy_L1) { isBuyReady = true; b1=b2=b3=b4=false; }
   if(isBuyReady && InpAllowBuy) {
      if(!b1 && currRSI >= InpBuy_L1) { if(OpenSpecificOrder(M_B1, "Buy", InpBuy_L1_SL, InpBuy_L1_TP)) b1=true; }
      if(!b2 && currRSI >= InpBuy_L2) { if(OpenSpecificOrder(M_B2, "Buy", InpBuy_L2_SL, InpBuy_L2_TP)) b2=true; }
      if(!b3 && currRSI >= InpBuy_L3) { if(OpenSpecificOrder(M_B3, "Buy", InpBuy_L3_SL, InpBuy_L3_TP)) b3=true; }
      if(!b4 && currRSI >= InpBuy_L4) { if(OpenSpecificOrder(M_B4, "Buy", InpBuy_L4_SL, InpBuy_L4_TP)) { b4=true; isBuyReady=false; } }
      if(currRSI >= InpBuy_BE_Zone) isBuyReady = false;
   }

   // --- SELL LOGIC ---
   if(currRSI > InpSell_L1) { isSellReady = true; s1=s2=s3=s4=false; }
   if(isSellReady && InpAllowSell) {
      if(!s1 && currRSI <= InpSell_L1) { if(OpenSpecificOrder(M_S1, "Sell", InpSell_L1_SL, InpSell_L1_TP)) s1=true; }
      if(!s2 && currRSI <= InpSell_L2) { if(OpenSpecificOrder(M_S2, "Sell", InpSell_L2_SL, InpSell_L2_TP)) s2=true; }
      if(!s3 && currRSI <= InpSell_L3) { if(OpenSpecificOrder(M_S3, "Sell", InpSell_L3_SL, InpSell_L3_TP)) s3=true; }
      if(!s4 && currRSI <= InpSell_L4) { if(OpenSpecificOrder(M_S4, "Sell", InpSell_L4_SL, InpSell_L4_TP)) { s4=true; isSellReady=false; } }
      if(currRSI <= InpSell_BE_Zone) isSellReady = false;
   }

   ManageRSI_Protections(currRSI);
   ManageProfitProtections();
}

void ManageProfitProtections() {
   for(int i = PositionsTotal() - 1; i >= 0; i--) {
      ulong ticket = PositionGetTicket(i);
      if(PositionSelectByTicket(ticket)) {
         long m = PositionGetInteger(POSITION_MAGIC);
         if(m < 987000 || m > 989000) continue;

         double open = PositionGetDouble(POSITION_PRICE_OPEN);
         double cur = PositionGetDouble(POSITION_PRICE_CURRENT);
         double sl = PositionGetDouble(POSITION_SL);
         int type = (int)PositionGetInteger(POSITION_TYPE);
         double pips = (type == POSITION_TYPE_BUY) ? (cur - open) : (open - cur);
         pips /= (pipsToPoints * _Point);

         int targetBE = (m == M_B1 || m == M_S1) ? InpBE_Main_Pips : InpBE_Sub_Pips;

         if(pips >= targetBE) {
            double nSL = (type == POSITION_TYPE_BUY) ? (open + InpBE_Add_Pips * pipsToPoints * _Point) : (open - InpBE_Add_Pips * pipsToPoints * _Point);
            if(type == POSITION_TYPE_BUY && (sl < open)) trade.PositionModify(ticket, nSL, PositionGetDouble(POSITION_TP));
            if(type == POSITION_TYPE_SELL && (sl > open || sl == 0)) trade.PositionModify(ticket, nSL, PositionGetDouble(POSITION_TP));
         }
      }
   }
}

void ManageRSI_Protections(double rsi) {
   for(int i = PositionsTotal() - 1; i >= 0; i--) {
      ulong ticket = PositionGetTicket(i);
      if(PositionSelectByTicket(ticket)) {
         long m = PositionGetInteger(POSITION_MAGIC);
         double open = PositionGetDouble(POSITION_PRICE_OPEN);
         int type = (int)PositionGetInteger(POSITION_TYPE);
         double curTP = PositionGetDouble(POSITION_TP);
         double nSL = (type == POSITION_TYPE_BUY) ? (open + InpBE_Add_Pips * pipsToPoints * _Point) : (open - InpBE_Add_Pips * pipsToPoints * _Point);

         // Buy RSI BE
         if(m >= 987010 && m <= 987025 && rsi >= InpBuy_BE_Zone) {
            if(PositionGetDouble(POSITION_SL) < open) trade.PositionModify(ticket, nSL, curTP);
         }
         // Sell RSI BE
         if(m >= 988075 && m <= 988090 && rsi <= InpSell_BE_Zone) {
            if(PositionGetDouble(POSITION_SL) > open || PositionGetDouble(POSITION_SL) == 0) trade.PositionModify(ticket, nSL, curTP);
         }
      }
   }
}

bool OpenSpecificOrder(int magic, string side, int sl_pips, int tp_pips) {
   double balance = AccountInfoDouble(ACCOUNT_BALANCE);
   double lot = MathFloor(((balance / InpBaseBalance) * InpBaseLot) / SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_STEP)) * SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_STEP);
   lot = MathMax(lot, SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_MIN)); lot = MathMin(lot, InpMaxLot);
   trade.SetExpertMagicNumber(magic);

   if(side == "Buy") {
      double price = SymbolInfoDouble(_Symbol, SYMBOL_ASK);
      double sl = price - sl_pips * pipsToPoints * _Point;
      double tp = price + tp_pips * pipsToPoints * _Point;
      return trade.Buy(lot, _Symbol, price, sl, tp);
   } else {
      double price = SymbolInfoDouble(_Symbol, SYMBOL_BID);
      double sl = price + sl_pips * pipsToPoints * _Point;
      double tp = price - tp_pips * pipsToPoints * _Point;
      return trade.Sell(lot, _Symbol, price, sl, tp);
   }
}
