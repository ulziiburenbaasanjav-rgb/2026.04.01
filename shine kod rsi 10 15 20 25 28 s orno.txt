#property strict
#include <Trade\Trade.mqh>

input group "=== RSI Түвшин ба Тохиргоо ==="
input int     InpRSI_Period   = 14;   
input double  InpLevel_Reset  = 10.0;  
input double  InpLevel_15     = 15.0;  
input double  InpLevel_20     = 20.0;  
input double  InpLevel_25     = 25.0;  
input double  InpLevel_30     = 30.0;  

input group "=== Ашиг ба Алдагдал (Pips) ==="
input int     InpTP_Final     = 150;   
input int     InpSL_Initial   = 100;   
input int     InpBE_Level10   = 100;   // 10-ын оролтын BE ашиг
input int     InpBE_Level15_25= 50;    // 15, 20, 25-ын BE ашиг

input group "=== Эрсдэлийн Удирдлага ==="
input double InpBaseBalance   = 50.0;
input double InpBaseLot       = 0.03;
input double InpMaxLot        = 10.0;

#define MAGIC_10 987010
#define MAGIC_15 987015
#define MAGIC_20 987020
#define MAGIC_25 987025

CTrade  trade;
int     handleRSI;
double  pipsToPoints;
bool    isReady = false; 
bool    l10_done = false, l15_done = false, l20_done = false, l25_done = false;

int OnInit()
{
   pipsToPoints = (_Digits == 3 || _Digits == 5) ? 10.0 : 1.0;
   handleRSI = iRSI(_Symbol, PERIOD_M5, InpRSI_Period, PRICE_CLOSE);
   return(INIT_SUCCEEDED);
}

void OnTick()
{
   double rsiBuffer[];
   ArraySetAsSeries(rsiBuffer, true);
   if(CopyBuffer(handleRSI, 0, 0, 2, rsiBuffer) < 2) return;
   
   double currRSI = rsiBuffer[0];

   // 1. RESET LOGIC
   if(currRSI < InpLevel_Reset) 
   {
      isReady = true;
      l10_done = false; l15_done = false; l20_done = false; l25_done = false;
   }

   // 2. ENTRY LOGIC
   if(isReady)
   {
      if(!l10_done && currRSI >= InpLevel_Reset) { if(OpenOrder(MAGIC_10)) l10_done = true; }
      if(!l15_done && currRSI >= InpLevel_15)    { if(OpenOrder(MAGIC_15)) l15_done = true; }
      if(!l20_done && currRSI >= InpLevel_20)    { if(OpenOrder(MAGIC_20)) l20_done = true; }
      if(!l25_done && currRSI >= InpLevel_25)    { if(OpenOrder(MAGIC_25)) { l25_done = true; isReady = false; } }
      
      // 30 давбал 25-ын оролтыг нөхөж нээх
      if(!l25_done && currRSI >= InpLevel_30)    { if(OpenOrder(MAGIC_25)) { l25_done = true; isReady = false; } }
   }

   // 3. RSI 30 ХҮРВЭЛ ШУУД BE БОЛГОЖ TP ТАВИХ
   Update_TP_and_BE_at_30(currRSI);

   // 4. RSI 30 ХҮРЭЭГҮЙ БАЙСАН Ч АШГААР НЬ BE БОЛГОХ
   ManageRiskByProfit();
}

// RSI 30 давмагц ажиллах функц
void Update_TP_and_BE_at_30(double rsi)
{
   if(rsi < InpLevel_30) return;

   for(int i = PositionsTotal() - 1; i >= 0; i--)
   {
      ulong ticket = PositionGetTicket(i);
      if(PositionSelectByTicket(ticket))
      {
         long m = PositionGetInteger(POSITION_MAGIC);
         if(m >= 987010 && m <= 987025)
         {
            double open = PositionGetDouble(POSITION_PRICE_OPEN);
            double sl = PositionGetDouble(POSITION_SL);
            double new_tp = open + InpTP_Final * pipsToPoints * _Point;

            // RSI 30-д хүрсэн бол ашиг хамаагүй шууд BE болгож, TP-г 150 дээр зооно
            if(sl < open || PositionGetDouble(POSITION_TP) == 0)
               trade.PositionModify(ticket, open, new_tp);
         }
      }
   }
}

// Ашигт хүрвэл BE болгох функц (RSI 30 хүрэхийг хүлээхгүй)
void ManageRiskByProfit()
{
   for(int i = PositionsTotal() - 1; i >= 0; i--)
   {
      ulong ticket = PositionGetTicket(i);
      if(PositionSelectByTicket(ticket))
      {
         long m = PositionGetInteger(POSITION_MAGIC);
         if(m < 987010 || m > 987025) continue;

         double open = PositionGetDouble(POSITION_PRICE_OPEN);
         double cur = PositionGetDouble(POSITION_PRICE_CURRENT);
         double sl = PositionGetDouble(POSITION_SL);
         double profitPips = (cur - open) / (pipsToPoints * _Point);

         // 10-ын оролт 100 pip ашигт хүрвэл
         if(m == MAGIC_10 && profitPips >= InpBE_Level10 && sl < open)
            trade.PositionModify(ticket, open, PositionGetDouble(POSITION_TP));

         // 15, 20, 25-ын оролтууд 50 pip ашигт хүрвэл
         if((m == MAGIC_15 || m == MAGIC_20 || m == MAGIC_25) && profitPips >= InpBE_Level15_25 && sl < open)
            trade.PositionModify(ticket, open, PositionGetDouble(POSITION_TP));
      }
   }
}

bool OpenOrder(int magic)
{
   double balance = AccountInfoDouble(ACCOUNT_BALANCE);
   double lot = MathFloor(((balance / InpBaseBalance) * InpBaseLot) / SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_STEP)) * SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_STEP);
   lot = MathMax(lot, SymbolInfoDouble(_Symbol, SYMBOL_VOLUME_MIN));
   lot = MathMin(lot, InpMaxLot);

   double price = SymbolInfoDouble(_Symbol, SYMBOL_ASK);
   double sl_val = price - InpSL_Initial * pipsToPoints * _Point;
   
   trade.SetExpertMagicNumber(magic);
   return trade.Buy(lot, _Symbol, price, sl_val, 0);
}
